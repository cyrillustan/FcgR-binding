1function [L] = StoneMod(logR, Kd, v, logKx, L0, biCoefMat)
 2    %Returns the number of mutlivalent ligand bound to a cell with 10^logR
 3    %receptors, granted each epitope of the ligand binds to the receptor
 4    %kind in question with dissociation constant Kd and cross-links with
 5    %other receptors with crosslinking constant Kx = 10^logKx. All
 6    %equations derived from Stone et al. (2001). Assumed that ligand is at
 7    %saturating concentration L0 = 7e-8 M, which is as it is (approximately)
 8    %for TNP-4-BSA in Lux et al. (2013).
 9    
10    Kx = 10^logKx;
11    R = 10^logR;
12    
13    %Vector of binomial coefficients
14    biCoefVec = biCoefMat(1:v,v)';
15    Req = 10^ReqFuncSolver(R,Kd,L0,v,Kx);
16    
17    %Calculate L, according to equations 1 and 7
18    L = sum(biCoefVec.*(Kx.^([1:v]-1)).*(L0/Kd*(Req.^[1:v])));
19end
