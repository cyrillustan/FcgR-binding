1function logSqrErr = NormalError2(Rtot, KdMat, mfiAdjMean, tnpbsa,...
 2    meanPerCond, biCoefMat, whichR)
 3
 4    sigCoef = 10^Rtot(7);
 5    logKxcoef = Rtot(2);
 6    logSqrErrMat = zeros(24,8);
 7    logR = Rtot(1);
 8    for j = 1:2
 9        v = Rtot(4+j);
10        c = 10^Rtot(2+j);
11        L0 = tnpbsa(j);
12        for k = 1:4
13            Kd = KdMat(whichR,k);
14            Kx = Kd*10^logKxcoef;
15            logKx = log10(Kx);
16            MFI = c*StoneMod(logR,Kd,v,logKx,L0,biCoefMat);
17            temp = mfiAdjMean(4*(whichR-1)+k,(4*(j-1)+1):(4*(j-1)+4));
18            mean = meanPerCond(4*(whichR-1)+k,j);
19            for l = 1:4
20                logSqrErrMat(4*(whichR-1)+k,4*(j-1)+l) = ...
21                    pseudoNormlike(temp(l),MFI,...
22                    sigCoef*mean);
23            end
24        end
25    end
26    logSqrErr = nansum(nansum(logSqrErrMat));
27    if logSqrErr < -realmax/(1e20)
28        logSqrErr = -realmax/(1e20);
29    end
30end
