21function c = ReqFuncSolver(R, kdi, Li, vi, kx)
22    %%%This function returns the point at which function fun equals zero
23    %%%using the bisection algorithm. The closest a and b will converge to
24    %%%in the algorithm is a distance 1e-12 apart.
25    
26    viLikdi = vi*Li/kdi;
27    
28    a = -20;
29    b = log10(R);
30    
31    bVal = fun(b, R, vi, kx, viLikdi);
32    cVal = fun(a, R, vi, kx, viLikdi);
33    
34    % Is there no root within the interval?
35    if bVal*cVal > 0
36        c = 1000;
37        return;
38    end
39    
40    %In the case that (b - a > 1e-4 || abs(cVal) > 1e-4) == 1 to begin
41    %with; only implemented for MATLAB Coder
42    c = 1000;
43    %Commence algorithm
44    while ((b - a > 1e-4) && (abs(cVal) > 1e-4))
45        c = (a+b)/2;
46        cVal = fun(c, R, vi, kx, viLikdi);
47        
48        if cVal*bVal >= 0
49            b = c;
50            bVal = cVal;
51        else
52            a = c;
53        end
54    end
55end
