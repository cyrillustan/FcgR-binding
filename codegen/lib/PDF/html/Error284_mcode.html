1function [J, mfiExp, mfiExpPre] = Error(Rtot, kd, mfiAdjMean4, mfiAdjMean26, v, biCoefMat, tnpbsa)
 2
 3    Rtot = 10.^Rtot;
 4    kx = Rtot(7);
 5    L = tnpbsa;
 6    
 7    %Preallocating space for the Req values (according to Stone et al.) for
 8    %each combination of IgG and FcgR per flavor of TNP-X-BSA.
 9    Req4 = zeros(6,4);
10    Req26 = zeros(6,4);
11
12%     ReqFunc = @(Reqi, R, kdi, Li, vi) R - Reqi*(1+vi*Li/kdi*(1+kx*Reqi)^(vi-1));
13
14    %Finding Req values by means of bisection algorithm
15    for j = 1:6
16        for k = 1:4
17%             Req4(j,k) = bisection(@(x) ReqFunc(10^x,Rtot(j),kd(j,k),L(1),v(1),kx),-5,5,1e-10);
18%             Req26(j,k) = bisection(@(x) ReqFunc(10^x,Rtot(j),kd(j,k),L(2),v(2),kx),-5,5,1e-10);
19            Req4(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(1), v(1), kx);
20            Req26(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(2), v(2), kx);
21            if isnan(Req4(j,k))
22                Req4(j,k) = -Inf;
23            end
24            if isnan(Req26(j,k))
25                Req26(j,k) = -Inf;
26            end
27        end
28    end
29    %Preventing errors in global optimization due to failure of the
30    %above local solver
31    if max(max(isnan(Req4))) || max(max(isnan(Req26))) || max(max(~isreal(Req4))) || max(max(~isreal(Req4)))
32        J = 1E6;
33        mfiExp = [];
34        mfiExpPre = [];
35        return;
36    end
37    
38    Req4 = 10.^Req4;
39    Req26 = 10.^Req26;
40    
41%   R is a seven-dimensional vector whose first six elements are
42%   expression levels of FcgRIA, FcgRIIA-Arg, etc. and whose seventh
43%   element is kx (see RegressionAndResiduals.m). kd is a 6 X 4
44%   matrix whose elements represent the Kd values associated with each
45%   flavor of immunoglobulin per flavor of receptor. tnpbsa is derived in
46%   loadData.m, as is mfiAdjMean; v is the valency of the TNP-X-BSA.
47%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
48%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
49
50    %Create a matrix of coefficients of the form:
51    %v!/((v-i)!*i!)*10^(kx+i-1)*tnpbsa
52    %for all i from 1 to v for all v from 1 to 10
53    
54    CoefVec4 = biCoefMat(1:v(1),v(1)) .* Rtot(size(Rtot,1)).^((1:v(1))'-1)*L(1);
55    CoefVec26 = biCoefMat(1:v(2),v(2)) .* Rtot(size(Rtot,1)).^((1:v(2))'-1)*L(2);
56    
57    %Creating matrix of expected MFIs; takes a few steps to do so
58    mfiExpPre4 = zeros(6,4);
59    mfiExpPre26 = zeros(6,4);
60    for j = 1:6
61        for k = 1:4
62            mfiExpPre4(j,k) = nansum(CoefVec4.*Req4(j,k).^(1:v(1))');
63            mfiExpPre26(j,k) = nansum(CoefVec26.*Req26(j,k).^(1:v(2))');
64        end
65    end
66    mfiExpPre4 = mfiExpPre4./kd;
67    mfiExpPre26 = mfiExpPre26./kd;
68    mfiExpPre = [mfiExpPre4 mfiExpPre26];
69        
70    mfiExp4 = zeros(24,4);
71    mfiExp26 = zeros(24,4);
72    for j = 1:6
73        for k = 1:4
74            mfiExp4((4*(j-1)+k),:) = mfiExpPre4(j,k)*ones(1,4);
75            mfiExp26((4*(j-1)+k),:) = mfiExpPre26(j,k)*ones(1,4);
76        end
77    end
78    %Expected MFIs
79    mfiExp = [mfiExp4, mfiExp26];
80    
81    %Error
82    J = nansum(nansum([(mfiExp4 - mfiAdjMean4).^2, (mfiExp26 - mfiAdjMean26).^2]));
83    %%%REMOVE
84    ErrorStruct = struct('J',J,'mfiExp',mfiExp,'mfiExpPre',mfiExpPre);
85end
