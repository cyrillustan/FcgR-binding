1function [logprob] = PDF(x, kd, mfiAdjMean, biCoefMat, tnpbsa, meanPerCond, stdPerCond)
 2    %Find residuals for the model granted current parameters
 3    
 4    %mfiExpPre is a 6x8 matrix which includes the predicted value by the
 5    %given parameter fit for each combination of FcgR, IgG, and valency. It
 6    %is the concatenation of matrices mfiExpPre4 and mfiExpPre26; see
 7    %Error.m for their definiton
 8    [~,~,mfiExpPre] = ErrorAvidityChange(x',kd,mfiAdjMean,biCoefMat,tnpbsa);
 9    
10    %Check to see that for the parameter fit there exist expected values
11    %for the data (see Error.m lines 23 through 28)
12    if mfiExpPre(1,1) == -1 || max(x) > 8
13        logprob = -Inf;
14        return
15    end
16    
17    %Create a matrix which includes the log probability of the model being
18    %chosen for each combination of data from one FcgR, one IgG, and one
19    %valency
20    logprob = 0;
21    for j = 1:6
22        for k = 1:4
23            for l = 1:2
24                 logprob = logprob + pseudoNormlike(mfiExpPre(j,4*(l-1)+k), ...
25                     meanPerCond(4*(j-1)+k,l),stdPerCond(4*(j-1)+k,l));
26            end
27        end
28    end
29end
