29function logprob = pseudoNormlike(x,mu,sigma)
30    %To replace normlike in the function PDF; while normlike returns
31    %negated log probabilities, this function returns log probabilities as
32    %they are.
33    z = (x - mu) / sigma;
34    logprob = -.5 * z^2 - log(sqrt(2.*pi).*sigma);
35end
