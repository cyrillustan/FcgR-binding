57function val = twotailexprnd(inversestd)
58    % inversestd represents the reciprocal of the standard deviation of the
59    % corresponding exponential distribution, which happens to be equal to
60    % the mean of the corresponding exponential distribution
61    val = exprnd(inversestd);
62    temp = randi(2);
63    if temp == 1
64        val = -val;
65    end
66end
