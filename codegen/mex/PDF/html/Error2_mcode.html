1function [J, mfiExp, mfiExpPre] = Error(Rtot, kd, mfiAdjMean4, mfiAdjMean26, v, biCoefMat, tnpbsa)
 2    
 3    % If error is called with Rtot being a single value, assume we want to
 4    % have constant expression across all the receptors
 5    if numel(Rtot) == 1
 6        Rtot = Rtot * ones(1,7);
 7    end
 8    
 9    Rtot = 10.^Rtot;
10    kx = Rtot(7);
11    L = tnpbsa;
12    
13    %Preallocating space for the Req values (according to Stone et al.) for
14    %each combination of IgG and FcgR per flavor of TNP-X-BSA.
15    Req4 = zeros(6,4);
16    Req26 = zeros(6,4);
17
18    %Finding Req values by means of bisection algorithm
19    for j = 1:6
20        for k = 1:4
21            Req4(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(1), v(1), kx);
22            Req26(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(2), v(2), kx);
23        end
24    end
25    %Preventing errors in global optimization due to failure of the
26    %above local solver
27    if max(max(Req4 == 1000)) || max(max(Req26 == 1000))
28        J = 1E8;
29        mfiExpPre = [];
30        mfiExp = [];
31        return;
32    end
33    
34    Req4 = 10.^Req4;
35    Req26 = 10.^Req26;
36    
37%   R is a one-dimensional vector whose first six elements are
38%   expression levels of FcgRIA, FcgRIIA-Arg, etc. and whose seventh
39%   element is kx (see RegressionAndResiduals.m). kd is a 6 X 4
40%   matrix whose elements represent the Kd values associated with each
41%   flavor of immunoglobulin per flavor of receptor. tnpbsa is derived in
42%   loadData.m, as is mfiAdjMean; v is the valency of the TNP-X-BSA.
43%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
44%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
45
46    %Create a matrix of coefficients of the form:
47    %v!/((v-i)!*i!)*10^(kx+i-1)*tnpbsa
48    %for all i from 1 to v for all v from 1 to 10
49    CoefVec4 = biCoefMat(1:v(1),v(1)) .* Rtot(size(Rtot,1)).^((1:v(1))'-1)*L(1);
50    CoefVec26 = biCoefMat(1:v(2),v(2)) .* Rtot(size(Rtot,1)).^((1:v(2))'-1)*L(2);
51    
52    %Creating matrix of expected MFIs; takes a few steps to do so
53    mfiExpPre4 = zeros(6,4);
54    mfiExpPre26 = zeros(6,4);
55    for j = 1:6
56        for k = 1:4
57            mfiExpPre4(j,k) = nansum(CoefVec4.*Req4(j,k).^(1:v(1))');
58            mfiExpPre26(j,k) = nansum(CoefVec26.*Req26(j,k).^(1:v(2))');
59        end
60    end
61    mfiExpPre4 = mfiExpPre4./kd;
62    mfiExpPre26 = mfiExpPre26./kd;
63    mfiExpPre = [mfiExpPre4 mfiExpPre26];
64        
65    mfiExp4 = zeros(24,4);
66    mfiExp26 = zeros(24,4);
67    for j = 1:6
68        for k = 1:4
69            mfiExp4((4*(j-1)+k),:) = mfiExpPre4(j,k)*ones(1,4);
70            mfiExp26((4*(j-1)+k),:) = mfiExpPre26(j,k)*ones(1,4);
71        end
72    end
73    %Expected MFIs
74    mfiExp = [mfiExp4, mfiExp26];
75    
76    %Error
77    J = nansum(nansum([(mfiExp4 - mfiAdjMean4).^2, (mfiExp26 - mfiAdjMean26).^2]));
78end
