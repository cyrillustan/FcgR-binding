1% function [J, mfiExp, mfiExpPre] = Error(Rtot, kd, mfiAdjMean4, mfiAdjMean26, v, biCoefMat, tnpbsa)
 2function ErrorStruct = Error(Rtot, kd, mfiAdjMean4, mfiAdjMean26, v, biCoefMat, tnpbsa)
 3    Rtot = 10.^Rtot;
 4    kx = Rtot(7);
 5    L = tnpbsa;
 6    
 7    %Preallocating space for the Req values (according to Stone et al.) for
 8    %each combination of IgG and FcgR per flavor of TNP-X-BSA.
 9    Req4 = zeros(6,4);
10    Req26 = zeros(6,4);
11
12%     ReqFunc = @(Reqi, R, kdi, Li, vi) R - Reqi*(1+vi*Li/kdi*(1+kx*Reqi)^(vi-1));
13
14    %Finding Req values by means of bisection algorithm
15    for j = 1:6
16        for k = 1:4
17%             Req4(j,k) = bisection(@(x) ReqFunc(10^x,Rtot(j),kd(j,k),L(1),v(1),kx),-5,5,1e-10);
18%             Req26(j,k) = bisection(@(x) ReqFunc(10^x,Rtot(j),kd(j,k),L(2),v(2),kx),-5,5,1e-10);
19            Req4(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(1), v(1), kx, 1e-5, 1e5, 1e-10);
20            Req26(j,k) = ReqFuncSolver(Rtot(j), kd(j,k), L(2), v(2), kx, 1e-5, 1e5, 1e-10);
21            if isnan(Req4(j,k))
22                Req4(j,k) = -Inf;
23            end
24            if isnan(Req26(j,k))
25                Req26(j,k) = -Inf;
26            end
27        end
28    end
29    %Preventing errors in global optimization due to failure of the
30    %above local solver
31    if max(max(isnan(Req4))) || max(max(isnan(Req26)))
32        J = 1E6;
33        mfiExp = [];
34        mfiExpPre = [];
35        ErrorStruct = struct('J',J,'mfiExp',mfiExp,'mfiExpPre',mfiExpPre);
36        return;
37    end
38    
39    Req4 = 10.^Req4;
40    Req26 = 10.^Req26;
41    
42%   R is a seven-dimensional vector whose first six elements are
43%   expression levels of FcgRIA, FcgRIIA-Arg, etc. and whose seventh
44%   element is kx (see RegressionAndResiduals.m). kd is a 6 X 4
45%   matrix whose elements represent the Kd values associated with each
46%   flavor of immunoglobulin per flavor of receptor. tnpbsa is derived in
47%   loadData.m, as is mfiAdjMean; v is the valency of the TNP-X-BSA.
48%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
49%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
50
51    %Create a matrix of coefficients of the form:
52    %v!/((v-i)!*i!)*10^(kx+i-1)*tnpbsa
53    %for all i from 1 to v for all v from 1 to 10
54    
55    CoefVec4 = biCoefMat(1:v(1),v(1)) .* Rtot(size(Rtot,1)).^((1:v(1))'-1)*L(1);
56    CoefVec26 = biCoefMat(1:v(2),v(2)) .* Rtot(size(Rtot,1)).^((1:v(2))'-1)*L(2);
57    
58    %Creating matrix of expected MFIs; takes a few steps to do so
59    mfiExpPre4 = zeros(6,4);
60    mfiExpPre26 = zeros(6,4);
61    for j = 1:6
62        for k = 1:4
63            mfiExpPre4(j,k) = nansum(CoefVec4.*Req4(j,k).^(1:v(1))');
64            mfiExpPre26(j,k) = nansum(CoefVec26.*Req26(j,k).^(1:v(2))');
65        end
66    end
67    mfiExpPre4 = mfiExpPre4./kd;
68    mfiExpPre26 = mfiExpPre26./kd;
69    mfiExpPre = [mfiExpPre4 mfiExpPre26];
70        
71    mfiExp4 = zeros(24,4);
72    mfiExp26 = zeros(24,4);
73    for j = 1:6
74        for k = 1:4
75            mfiExp4((4*(j-1)+k),:) = mfiExpPre4(j,k)*ones(1,4);
76            mfiExp26((4*(j-1)+k),:) = mfiExpPre26(j,k)*ones(1,4);
77        end
78    end
79    %Expected MFIs
80    mfiExp = [mfiExp4, mfiExp26];
81    
82    %Error
83    J = nansum(nansum([(mfiExp4 - mfiAdjMean4).^2, (mfiExp26 - mfiAdjMean26).^2]));
84    %%%REMOVE
85    ErrorStruct = struct('J',J,'mfiExp',mfiExp,'mfiExpPre',mfiExpPre);
86end
