 2    %Find residuals for the model granted current parameters
 3    
 4    %mfiExpPre is a 6x8 matrix which includes the predicted value by the
 5    %given parameter fit for each combination of FcgR, IgG, and valency. It
 6    %is the concatenation of matrices mfiExpPre4 and mfiExpPre26; see
 7    %Error.m for their definiton
 8    
 9    %TEMP
10    %ErrorStruct is a structured array with fields 'J', 'mfiExp', and

19    if isempty(mfiExpPre)
20        logprob = -Inf;

22    end
23    
24    %Create a matrix which includes the log probability of the model being

36            end

38    end
39    
40    %Sum up negative log probabilities and negate
41    logprob = -sum(sum(logprobMat));
42end
