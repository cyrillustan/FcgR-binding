1function x = ReqFuncSolver(R, kdi, Li, vi, kx)
 2    %%%This function returns the point at which function fun equals zero
 3    %%%using the bisection algorithm. The closest a and b will converge to
 4    %%%in the algorithm is a distance 1e-12 apart.
 5    
 6    %Check that the first point on the interval is less than the second, 
 7    %that fun(a)*fun(b) is negative, and that error is positive. If not, 
 8    %return an imaginary number.
 9    
10    a = -5;
11    b = 5;
12    error = -10;
13    
14    %Commence algorithm
15    while 1
16        c = (a+b)/2;
17        if abs(fun(c, R, kdi, Li, vi, kx)) < error
18            x = c;
19            return;
20        end
21        if fun(c, R, kdi, Li, vi, kx)*fun(b, R, kdi, Li, vi, kx) >= 0
22            b = c;
23        else
24            a = c;
25        end
26        %Returns a NaN if a and b come with 1e-12 of each other before
27        %convergence
28        if b-a < 1e-12
29            x = 6;
30            return
31        end
32    end
33end
