18function c = ReqFuncSolver(R, kdi, Li, vi, kx)
19    %%%This function returns the point at which function fun equals zero
20    %%%using the bisection algorithm. The closest a and b will converge to
21    %%%in the algorithm is a distance 1e-12 apart.
22    
23    a = -20;
24    b = log10(R);
25    
26    bVal = fun(b, R, kdi, Li, vi, kx);
27    cVal = fun(a, R, kdi, Li, vi, kx);
28    
29    % Is there no root within the interval?
30    if bVal*cVal > 0
31        c = 1000;
32        return;
33    end
34    
35    %In the case that (b - a > 1e-4 || abs(cVal) > 1e-4) == 1 to begin
36    %with; only implemented for MATLAB Coder
37    c = 1000;
38    %Commence algorithm
39    while b - a > 1e-4 || abs(cVal) > 1e-4
40        c = (a+b)/2;
41        cVal = fun(c, R, kdi, Li, vi, kx);
42        
43        if cVal*bVal >= 0
44            b = c;
45            bVal = cVal;
46        else
47            a = c;
48        end
49    end
50end
