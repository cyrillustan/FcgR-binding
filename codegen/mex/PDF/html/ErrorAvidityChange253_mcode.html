1function [J, mfiExp, mfiExpPre, mfiDiff] = ErrorAvidityChange(RtotTrue, Kd,...
 2    mfiAdjMean, biCoefMat, tnpbsa)
 3    %Treats the last two elements of RtotTrue as the effective avidities of
 4    %TNP-4-BSA and TNP-26-BSA respectively (as of June 27, 2016, these are
 5    %elements 10 and 11). Runs Error.m inputting these two avidities as a
 6    %two-dimensional column vector which is passed into Error as the vector
 7    %v.
 8    
 9    % If error is called with Rtot being a single value, assume we want to
10    % have constant expression across all the receptors
11    if numel(RtotTrue) == 6
12        RtotTrue = [RtotTrue(1) * ones(6,1); RtotTrue(2:6)];
13    end
14    
15    [J, mfiExp, mfiExpPre] = Error(RtotTrue(1:9),Kd,mfiAdjMean,ceil(RtotTrue(10:11)),...
16        biCoefMat,tnpbsa);
17    
18    mfiDiff = mfiExp - mfiAdjMean;
19end
