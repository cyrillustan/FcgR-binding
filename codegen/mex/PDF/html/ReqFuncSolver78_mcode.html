18function c = ReqFuncSolver(R, kdi, Li, vi, kx)
19    %%%This function returns the point at which function fun equals zero
20    %%%using the bisection algorithm. The closest a and b will converge to
21    %%%in the algorithm is a distance 1e-12 apart.
22    
23    viLikdi = vi*Li/kdi;
24    
25    a = -20;
26    b = log10(R);
27    
28    bVal = fun(b, R, vi, kx, viLikdi);
29    cVal = fun(a, R, vi, kx, viLikdi);
30    
31    % Is there no root within the interval?
32    if bVal*cVal > 0
33        c = 1000;
34        return;
35    end
36    
37    %In the case that (b - a > 1e-4 || abs(cVal) > 1e-4) == 1 to begin
38    %with; only implemented for MATLAB Coder
39    c = 1000;
40    %Commence algorithm
41    while b - a > 1e-4 || abs(cVal) > 1e-4
42        c = (a+b)/2;
43        cVal = fun(c, R, vi, kx, viLikdi);
44        
45        if cVal*bVal >= 0
46            b = c;
47            bVal = cVal;
48        else
49            a = c;
50        end
51    end
52end
