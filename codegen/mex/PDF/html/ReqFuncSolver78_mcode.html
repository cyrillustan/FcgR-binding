19function c = ReqFuncSolver(R, kdi, Li, vi, kx)
20    %%%This function returns the point at which function fun equals zero
21    %%%using the bisection algorithm. The closest a and b will converge to
22    %%%in the algorithm is a distance 1e-12 apart.
23    
24    viLikdi = vi*Li/kdi;
25    
26    a = -20;
27    b = log10(R);
28    
29    bVal = fun(b, R, vi, kx, viLikdi);
30    cVal = fun(a, R, vi, kx, viLikdi);
31    
32    % Is there no root within the interval?
33    if bVal*cVal > 0
34        c = 1000;
35        return;
36    end
37    
38    %In the case that (b - a > 1e-4 || abs(cVal) > 1e-4) == 1 to begin
39    %with; only implemented for MATLAB Coder
40    c = 1000;
41    %Commence algorithm
42    while b - a > 1e-4 || abs(cVal) > 1e-4
43        c = (a+b)/2;
44        cVal = fun(c, R, vi, kx, viLikdi);
45        
46        if cVal*bVal >= 0
47            b = c;
48            bVal = cVal;
49        else
50            a = c;
51        end
52    end
53end
