1function L = StoneSolver(Rtot,Kx,v,Kd,L0,biCoefMat)    
 2    %Using the information given, finds the sum L presented in Equation 7
 3    %in Stone. In this context, all inputs save biCoefMat are scalars.
 4    
 5    %Solve for Req, as described in Equation 2 in Stone
 6    Req = ReqFuncSolver(Rtot, Kd, L0, v, Kx);
 7    %Check for error output from ReqFuncSolver
 8    if Req == 1000
 9        L = -1;
10        return
11    end
12    %Convert from logarithmic scale
13    Req = 10^Req;
14    
15    L = sum(L0/Kd*(biCoefMat(1:v,v)'.*(Kx.^([1:v]-1))).*Req.^[1:v]);
16end
