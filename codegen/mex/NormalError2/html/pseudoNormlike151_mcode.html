27function logprob = pseudoNormlike(x,mu,sigma)
28    %To replace normlike in the function PDF; while normlike returns
29    %negated log probabilities, this function returns log probabilities as
30    %they are.
31    z = (x - mu) / sigma;
32    logprob = -.5 * z^2 - log(sqrt(2.*pi).*sigma);
33end
