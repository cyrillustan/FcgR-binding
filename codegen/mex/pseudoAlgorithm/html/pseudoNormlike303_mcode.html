31function logprob = pseudoNormlike(x,mu,sigma)
32    %To replace normlike in the function PDF; while normlike returns
33    %negated log probabilities, this function returns log probabilities as
34    %they are.
35    z = (x - mu) / sigma;
36    logprob = -.5 * z^2 - log(sqrt(2.*pi).*sigma);
37end
