30function logprob = pseudoNormlike(x,mu,sigma)
31    %To replace normlike in the function PDF; while normlike returns
32    %negated log probabilities, this function returns log probabilities as
33    %they are.
34    z = (x - mu) / sigma;
35    logprob = -.5 * z^2 - log(sqrt(2.*pi).*sigma);
36end
